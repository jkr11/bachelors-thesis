\subsection{Definitions}

Lets list some definitions of Tensor. A Tensor $T$ can be regarded as an element of a tensor product space
\[
  T \in \bigotimes_{j \in J} V^j
\]
and we say $T$ is a $\left\vert J \right\vert$-legged tensor.

Given finite dimensional Vector spaces $V$ and $W$, we have the identity that $V \otimes W^\ast \equiv Hom(W,V)$. Now, if we have a Tensor $T \in V_1 \otimes \cdots \otimes V_n$, and since for any Vector space $V$, $(V^\ast)^\ast = V$, we can peel off any number of $V_{i}$s, take their duals and represent $T$ as a map
\[
  T : V_{n}^\ast \otimes \cdots \otimes V_{k+1} ^\ast \to V_1 \otimes \cdots \otimes V_k 
\]
Note that we present the domain here in a contravariant fashion due to the contravariance of the dual.

Lets look at an example, let $t : V_1 \otimes V_2 \otimes W_{2}^\ast \otimes W_1^\ast$. We can represent as an operator in dirac notation:
\[
  t = \sum_{i_1,i_2,j_1,j_2}^{} T_{j1 j2 i1 i2}  \ket{v_{i_1}} \ket{v_{i_2}} \bra{w_{j_1}} \bra{w_{j_2}}
\] 

Graphically, we represent a general tensor
\[
  T : (W_{n}^{\ast} \otimes \cdots \otimes W_1) \to (V_1 \otimes V_2 \otimes \cdots \otimes V_n)
\]
as follows, where we choose incoming arrows for duals in the domain and outgoings for normal spaces, and on the bottom we choose incoming for duals and outgoing for normals.

\begin{lemma}\label{IsotopyHelper}
  For a vector space $W$, there is an isomorphism
  \[
    \bigoplus^m W \simeq \mathbb C^m \otimes W.
  \]
\end{lemma}
\begin{proof}
  Let ${e_i}_{i \in [m]}$ be the standard basis on $\mathbb{C}^m$. Let
  \[
    \Phi : \mathbb{C}^m \otimes W \to \oplus^m W
  \]
  be defined by
  \[
    \phi(e_i \otimes w) = (0, \ldots , w, \ldots 0  )
  \]
  and linear, thus
  \[
    \Phi\left(\sum_{i}^{} c_{i} e_{i} \otimes w_{i} \right) = (c_1 w_1, \ldots , c_m w_m ).
  \]
  is clearly bijective, and we have our desired isomorphism.
\end{proof}

\begin{lemma}\label{HomTensorFactorization}
Let $C$ be a $k$-linear category and let $D,E$ be finite-dimensional
$k$-vector spaces. For any objects $R,S\in C$, there is a natural
isomorphism
\[
\Hom_C(D\otimes R,E\otimes S)
\cong
\Hom_k(D,E)\otimes \Hom_C(R,S).
\]
\end{lemma}
\begin{proof}
Let $m=\dim_k D$ and $n=\dim_k E$. By unfolding the definitions and utilizing the properties of finite biproducts, we establish the result through a direct chain of natural isomorphisms:
$$
  \begin{aligned}
    \Hom_C(D\otimes R,E\otimes S)
    &\cong \Hom_C(R^{\oplus m},S^{\oplus n})
    && \text{by \ref{IsotopyHelper}} \\[1ex]
    &\cong \bigoplus_{i=1}^{m} \bigoplus_{j=1}^{n} \Hom_C(R,S)
    && \text{by additivity of $\Hom_C$} \\[1ex]
    &\cong k^{mn} \otimes \Hom_C(R,S)
    && \text{by \ref{IsotopyHelper}} \\[1ex]
    &\cong \Hom_k(D,E)\otimes \Hom_C(R,S)
    && \text{since $\Hom_k(D,E) \cong k^{mn}$}
  \end{aligned}
$$
\end{proof}

\begin{theorem}[Decomposition of Semisimple Categories]
  \label{DecompositionOfSemisimpleCategories}
  Let $C$ be a semisimple $k$-linear category with simple objects $\{R_a\}_{a\in L}$ and let $V_i, W_i$ be a collection of objects in $C$.
  Every morphism
  \[
    T\in \mathrm{Hom}_C\left(\bigotimes_i V_i, \bigotimes_i W_{i}\right)
  \]
  admits a decomposition
  \[
    T = \bigoplus_{\hat a,\hat b}
    \sum_\mu
    P^{(\hat a,\hat b,\mu)}
    \otimes
    Q^{(\hat a,\hat b,\mu)},
    \]
  where
  \[
    P^{(\hat a,\hat b,\mu)}
    \in
    \mathrm{Hom}
    \!\left(
    \bigotimes_i D_{i,a_i},
    \bigotimes_j E_{j,b_j}
    \right),
  \]
and
\[
Q^{(\hat a,\hat b,\mu)}
\in
\mathrm{Hom}_C(R_{\hat a},R_{\hat b}).
\]
\end{theorem}

\begin{proof}
Since $C$ is semisimple, every object decomposes as a finite direct sum
of simple objects. Thus, by Lemma~\ref{...} and the distributivity of the tensor product through the biproduct

\begin{equation}
  \label{TensorProductDecomposition}
  \begin{aligned}
\bigotimes_i V_i
    &\cong
      \bigotimes_i
      \left(
        \bigoplus_{a_i}
        D_{i,a_i}\otimes R_{a_i}
      \right)
\\
    &\cong
      \bigoplus_{\hat a}
      \left(
        \bigotimes_i D_{i,a_i}
      \right)
      \otimes
      \left(
        \bigotimes_i R_{a_i}
      \right).
  \\
  &\cong 
    \bigoplus_{\hat{a} } D_{\hat{a} } \otimes R_{\hat{a} }   
\end{aligned}
\end{equation}

Then

$$
  \begin{aligned}
    \Hom_C \left(
      \bigotimes_i V_i,
      \bigotimes_j W_j
    \right)
    &\cong
    \mathrm{Hom}_C
      \!\left(
      \bigoplus_{\hat a}
          D_{\hat a}\otimes R_{\hat a},
          \,
          \bigoplus_{\hat b}
          E_{\hat b}\otimes R_{\hat b}
          \right)
    && \text{by \eqref{TensorProductDecomposition}} \\[1ex]
    &\cong
    \bigoplus_{\hat a,\hat b}
    \mathrm{Hom}_C
    (
    D_{\hat a}\otimes R_{\hat a},
    E_{\hat b}\otimes R_{\hat b}
    ) && \text{by additivity of $Hom_C$} \\[1ex]
    &\cong
    \bigoplus_{\hat a,\hat b}
    \mathrm{Hom}(D_{\hat a},E_{\hat b})
      \otimes
    \mathrm{Hom}_C(R_{\hat a},R_{\hat b}).
    && \text{by \eqref{HomTensorFactorization}} 
  \end{aligned}
$$

Choosing a basis
\[
  \{Q^{(\hat a,\hat b,\mu)}\}_\mu
\]
of each space
\[
  \mathrm{Hom}_C(R_{\hat a},R_{\hat b}),
\]
every morphism $T$ can therefore be written uniquely as
\[
  T
  =
  \bigoplus_{\hat a,\hat b}
  \sum_\mu
  P^{(\hat a,\hat b,\mu)}
  \otimes
  Q^{(\hat a,\hat b,\mu)},
  \]
  with
  \[
  P^{(\hat a,\hat b,\mu)}
  \in
  \mathrm{Hom}(D_{\hat a},E_{\hat b}).
\]
\end{proof}

\begin{definition}[The Category $\Rep_k(G)$]
  Let $G$ be a group and $k$ a field. The \emph{category of linear representations of $G$ over $k$} consists of the following data:

  \begin{itemize}
    \item \textbf{Objects:} The objects $\Rep_k(G)$ are pairs $(V, \rho_V)$, where $V$ is a $k$-vector space and $\rho_V : G  \to \mathrm{GL}(V)$.  
    \item \textbf{Morphisms:} For any two objects $(V, \rho_V)$ and $(W, \rho_W)$ in $\Rep_k(G)$, a morphism from $V$ to $W$ is a $k$-linear transformation $f: V \to W$ that is $G$-invariant (also called an \textit{intertwiner}). That is, for every element $g \in G$, the following diagram commutes:
    \[
    \begin{tikzcd}
    V \arrow[r, "\rho_V(g)"] \arrow[d, "f"'] & V \arrow[d, "f"] \\
    W \arrow[r, "\rho_W(g)"']               & W
    \end{tikzcd}
    \]
    In algebraic terms, $f$ must satisfy the condition:
    \[ f \circ \rho_V(g) = \rho_W(g) \circ f \quad \forall g \in G \]
    The set of all such morphisms is denoted by $\mathrm{Hom}_G(V, W)$.
  \end{itemize}
  We can further endow $Rep_k(G)$ with a monoidal structure: For any two objects $(V,\rho_V), (W,\rho_W)$, their tensor product is in $\Rep_k(G)$ by the diagonal action
  \[
    \rho_{(V \otimes W)} (g) (v \otimes w) = \rho_V(g)v \otimes \rho_W(g)w
  \]
\end{definition}
